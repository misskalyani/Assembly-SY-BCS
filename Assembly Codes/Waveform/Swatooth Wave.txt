SWATOOTH WAVE NORMAL

ORG 00H
MOV A,#00H
L1:INC A
MOV P2,A
CJNE A,#0FFH,L1
SJMP L1


SWATOOTH WAVE WAVELENGTH CHANGE

ORG 00H
MOV A,#00H
L1:INC A
MOV P2,A
ACALL DELAY
CJNE A,#0FFH,L1
SJMP L1
DELAY:
MOV R0,#08H
J1:DJNZ R0,J1
RET


SWATOOTH WAVE AMPLITUDE CHANGE

ORG 00H
MOV A,#00H
L1:INC A
MOV P2,A
ACALL DELAY
ACALL DELAY
ACALL DELAY
CJNE A,#60H,L1
MOV A,#00H
SJMP L1
DELAY:
MOV R1,#05H
J1:DJNZ R1,J1
RET
