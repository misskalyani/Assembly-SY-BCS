HEXA TO DECIMAL

ORG 00H
MOV A,#10H
MOV B,#A6H
DIV AB
MOV R0,A
MOV R1,B
END
 
DECIMAL TO HEXA
 
ORG 00H
MOV R0,#07H
MOV R1,#32H
MOV A,R0
MOV B,R1
END

DECIMAL TO BINARY

ORG 00H
MOV R1,#O8H
MOV R2,#1OH
MOV A,#32D
MOV B,#O5D
BACK:DIV AB
MOV @R1,B
MOV B,#O5D
INC R1
DJNZ R1,BACK
END
